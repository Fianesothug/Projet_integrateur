création de table

CREATE DATABASE gestion;

USE gestion;
CREATE TABLE client (
    id INT AUTO_INCREMENT PRIMARY KEY,
    nom VARCHAR(100) NOT NULL,
    prenom VARCHAR(100) NOT NULL,
    sexe VARCHAR(10),
    pays VARCHAR(100),
    numero VARCHAR(20),
    email VARCHAR(150) UNIQUE,
    adresse TEXT,
    identifiant VARCHAR(100) UNIQUE,
    code VARCHAR(255)
);
